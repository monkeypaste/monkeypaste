90 public static bool IsValueBase64(this string str) {
91
92 if (string.IsNullOrWhiteSpace(str)) {
93 // host app convention checks
94 return false;
95 }
96 if (str.Length % 4 != 0) {
97 return false;
98 }
99 if (Regex.IsMatch(@"[^a-zA-Z0-9+/=]", str)) {
100 // str has non base64 char
101 return false;
102 }
103 try {
104 // If no exception is caught, then it is possibly a base64 encoded string
105 byte[] data = Convert.FromBase64String(str);
106 // The part that checks if the string was properly padded to the
107 // correct length was borrowed from d@anish's solution
108 return (str.Replace(" ", "").Length % 4 == 0);
109 }
110 catch (FormatException) {
111 // If exception is caught, then it is not a base64 encoded string
112
113 return false;
114 }
115 }
